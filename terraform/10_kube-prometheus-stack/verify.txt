# 10_kube-prometheus-stack — verification commands (run on bastion)

# Terraform apply finished
cd ~/tws-e-commerce-app_hackathon/terraform/10_kube-prometheus-stack
terraform show

# Helm release
helm list -n monitoring

# Namespace and pods
kubectl get ns monitoring
kubectl get pods -n monitoring

# Core components
kubectl get pods -n monitoring | grep prometheus
kubectl get pods -n monitoring | grep grafana
kubectl get pods -n monitoring | grep alertmanager

# Services
kubectl get svc -n monitoring

# Grafana admin password (default secret name from chart)
kubectl get secret -n monitoring -l app.kubernetes.io/name=grafana -o name
