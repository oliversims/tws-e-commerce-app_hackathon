# 07_alb-controller — verification commands (run on bastion)

# Helm release
helm list -n kube-system | grep load-balancer

# Controller pod running
kubectl get pods -n kube-system -l app.kubernetes.io/name=aws-load-balancer-controller

# IRSA on service account
kubectl get sa aws-load-balancer-controller -n kube-system -o jsonpath='{.metadata.annotations.eks\.amazonaws\.com/role-arn}'; echo
